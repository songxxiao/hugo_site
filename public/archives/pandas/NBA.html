ss =pd.read_csv("E:\\R_codes\\others\\nba\\Seasons_Stats.csv",header=0,index_col=None,encoding = 'utf-8')  
ss = ss[ss['Year']>=2017] # 2017
stats17 = ss[['Year','Player','Pos','Age','G','PER','MP','PTS','AST','TRB','TOV','BLK','STL']]
stats17.drop_duplicates(subset=['Player'],keep='first',inplace=True) #
stats17['MPG'] =stats17['MP']/stats17['G']
stats17['PPG'] =stats17['PTS']/stats17['G']
stats17['APG'] =stats17['AST']/stats17['G']
stats17['RPG'] =stats17['TRB']/stats17['G']
stats17['TOPG'] =stats17['TOV']/stats17['G']
stats17['BPG'] =stats17['BLK']/stats17['G']
stats17['SPG'] =stats17['STL']/stats17['G']
stats17['Pos'][stats17['Pos']=='PF-C'] = 'PF' #
#
stats_salary = pd.merge(stats17, salary_table)
